// üõ†Ô∏è MIDDLEWARE EXEMPLE - backend/src/middleware/subscription.js
// =============================================================================

const { Subscription, Plan } = require('../models');

/**
 * Middleware pour v√©rifier les limites du plan utilisateur
 */
const checkPlanLimits = (limitType) => {
  return async (req, res, next) => {
    try {
      // Si pas d'utilisateur connect√©, appliquer limites free
      if (!req.user) {
        req.userPlan = await Plan.findOne({ where: { slug: 'free' } });
        return next();
      }

      // R√©cup√©rer l'abonnement actuel
      const subscription = await Subscription.findOne({
        where: { 
          user_id: req.user.id, 
          status: ['active', 'trialing'] 
        },
        include: [{ model: Plan, as: 'plan' }]
      });

      if (!subscription) {
        // Pas d'abonnement = plan gratuit
        req.userPlan = await Plan.findOne({ where: { slug: 'free' } });
      } else {
        req.userPlan = subscription.plan;
        req.userSubscription = subscription;
      }

      // V√©rifier la limite sp√©cifique
      const limits = req.userPlan.limits || {};
      const limit = limits[limitType];

      if (limit !== undefined && limit !== -1) { // -1 = illimit√©
        // Ici on v√©rifierait l'usage actuel vs la limite
        // Par exemple pour daily_searches, on compterait les recherches du jour
        
        // Si limite d√©pass√©e
        if (req.currentUsage >= limit) {
          return res.status(429).json({
            error: 'Limite du plan atteinte',
            current_plan: req.userPlan.name,
            limit_type: limitType,
            current_usage: req.currentUsage,
            limit: limit,
            upgrade_url: '/plans',
            suggestions: {
              premium: 'Passez au plan Premium pour des limites plus √©lev√©es',
              pro: 'Le plan Pro offre un acc√®s illimit√©'
            }
          });
        }
      }

      next();
    } catch (error) {
      console.error('Erreur v√©rification limites plan:', error);
      res.status(500).json({ error: 'Erreur serveur' });
    }
  };
};

/**
 * Middleware pour tracker l'usage
 */
const trackUsage = (actionType) => {
  return async (req, res, next) => {
    // Tracker l'usage pour analytics et facturation
    if (req.user && req.userSubscription) {
      // Incr√©menter compteurs d'usage
      // Ceci pourrait √™tre fait de mani√®re asynchrone
      setImmediate(() => {
        // Logique de tracking
        console.log(`User ${req.user.id} performed ${actionType}`);
      });
    }
    next();
  };
};

module.exports = {
  checkPlanLimits,
  trackUsage
};

      error: 'Erreur de paiement',

      retry_url: '/api/payments/retry/' + err.payment_id
    });
  }
  
  if (err.type === 'subscription_error') {
    return res.status(403).json({
      error: 'Probl√®me d\'abonnement',
      message: err.message,
      upgrade_url: '/plans'
    });
  }
  
  next(err);
});
